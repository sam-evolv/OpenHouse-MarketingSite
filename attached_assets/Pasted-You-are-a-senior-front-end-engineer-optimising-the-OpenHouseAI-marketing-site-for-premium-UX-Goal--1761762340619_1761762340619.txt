2️⃣ Create Scroll Manager

In /lib/scrollManager.ts, initialise Lenis:

ts
Copy code
import Lenis from "@studio-freight/lenis";

export const initSmoothScroll = () => {
  const lenis = new Lenis({
    duration: 1.2,
    easing: (t) => Math.min(1, 1.001 - Math.pow(2, -10 * t)), // subtle decel
    smoothWheel: true,
    smoothTouch: true,
    direction: "vertical",
    gestureDirection: "vertical",
    touchMultiplier: 1.5,
  });

  function raf(time: number) {
    lenis.raf(time);
    requestAnimationFrame(raf);
  }
  requestAnimationFrame(raf);

  return lenis;
};
3️⃣ Integrate Globally

In app/layout.tsx (or app/page.tsx if using a single layout):

tsx
Copy code
"use client";
import { useEffect } from "react";
import { initSmoothScroll } from "@/lib/scrollManager";

export default function RootLayout({ children }: { children: React.ReactNode }) {
  useEffect(() => {
    const lenis = initSmoothScroll();
    return () => lenis.destroy();
  }, []);

  return <html lang="en"><body>{children}</body></html>;
}
